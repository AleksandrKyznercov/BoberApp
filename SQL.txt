CREATE TABLE Customer(
       ID_Customer      INTEGER AUTO_INCREMENT PRIMARY KEY,
       FIO				VARCHAR(300),
       passSerialNumber VARCHAR(20),
       passNumber       VARCHAR(20),
       vidan 	    	VARCHAR(550),
       adressProp		VARCHAR(500),
       adressFact 		VARCHAR(500),
       phone	    	VARCHAR(220)
) ENGINE = MYISAM;

CREATE TABLE Treaty(
       ID_Treaty        INTEGER AUTO_INCREMENT PRIMARY KEY,
       ID_Customer      INTEGER PRIMARY KEY,
       dateStart	 	DATETIME,
       dateEnd	        DATETIME,
	   commonCost		FLOAT,
	   Deposit			FLOAT,
       Status			VARCHAR(20)
) ENGINE = MYISAM;

CREATE TABLE Аction(
       ID_Action		INTEGER AUTO_INCREMENT PRIMARY KEY,
	   ID_Equipment		INTEGER PRIMARY KEY,
	   ID_Treaty		INTEGER PRIMARY KEY,
	   Cost				FLOAT,
	   RentTime			VARCHAR(20),
	   Deposit			FLOAT
) ENGINE = MYISAM;

CREATE TABLE Equipment(
       ID_Equipment		INTEGER AUTO_INCREMENT PRIMARY KEY,
       Name 			VARCHAR(220),
       SerialNumber		VARCHAR(220),
       Cost				INTEGER,
       Deposit			INTEGER,
	   Defenition		VARCHAR(550),
	   PriceFor4		INTEGER,
	   PriceFor8		INTEGER,
	   PriceFor24		INTEGER,
	   Remark 			VARCHAR(220),
	   Status 			VARCHAR(220)
) ENGINE = MYISAM;
