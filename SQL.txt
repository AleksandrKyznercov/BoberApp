CREATE TABLE Customer(
       ID_Customer      INTEGER AUTO_INCREMENT PRIMARY KEY,
       FIO			    VARCHAR(220),
       passSerialNumber VARCHAR(220),
       passNumber       VARCHAR(220),
       vidan 	    	VARCHAR(220),
       adressProp		VARCHAR(220),
       adressFact 		VARCHAR(220),
       phone	    	VARCHAR(220)
) ENGINE = MYISAM;

CREATE TABLE Treaty(
       ID_Treaty        INTEGER AUTO_INCREMENT PRIMARY KEY,
       ID_Customer      INTEGER,
       Start	 	    DATE,
       Stop	            INTEGER,
       Status			VARCHAR(20)
) ENGINE = MYISAM;

CREATE TABLE Аction(
       ID_Action		INTEGER AUTO_INCREMENT PRIMARY KEY,
	   ID_Equipment		INTEGER,
	   ID_Treaty		INTEGER,
	   Cost				FLOAT,
	   RentTime			VARCHAR(20)
	   
) ENGINE = MYISAM;

CREATE TABLE Equipment(
       ID_Equipment		INTEGER AUTO_INCREMENT PRIMARY KEY,
       Name 			VARCHAR(220),
       SerialNumber		VARCHAR(220),
       Cost				INTEGER,
       Deposit			INTEGER,
	   Defenition		VARCHAR(550),
	   PriceFor4		INTEGER,
	   PriceFor8		INTEGER,
	   PriceFor24		INTEGER,
	   Remark 			VARCHAR(220),
	   Status 			VARCHAR(220)
) ENGINE = MYISAM;
